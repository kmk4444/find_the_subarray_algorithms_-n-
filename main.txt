int o_n_kare(int * arr, int size){
	int number_of_sub_array = 0;
	int i, j;
	for(i=0; i< size -1;i++)
        for(j=size-1;j>i;j--)
        {
            if(arr[i] == 1 && arr[j] == 9)
                number_of_sub_array++;

        }

	return number_of_sub_array;
}

            number_of_one++;

            number_of_sub_array += number_of_one;
        }
	}

	return number_of_sub_array;
}
